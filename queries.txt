CREATE DATABASE DATAWAREHOUSE
USE DATAWAREHOUSE;

CREATE TABLE CONTACTOS(
ID INT PRIMARY KEY AUTO_INCREMENT,
NOMBRE VARCHAR(150) NOT NULL,
APELLIDO VARCHAR(150) NOT NULL,
MAIL VARCHAR(150) NOT NULL,
PAIS VARCHAR(150),
REGION VARCHAR(150),
COMPANIA VARCHAR(250) NOT NULL,
CARGO VARCHAR (250) NOT NULL,    
INTERES INT(3),
FOTO VARCHAR (250) NOT NULL,
CIUDAD VARCHAR (250) NOT NULL,
DIRECCION VARCHAR (250) NOT NULL,
TWITTER VARCHAR (250) NOT NULL,
PREFERENCIATW VARCHAR (250) NOT NULL,
FACEBOOK VARCHAR (250) NOT NULL,
PREFERENCIAFB VARCHAR (250) NOT NULL
);

SELECT * FROM CONTACTOS 

CREATE TABLE USUARIOS(	
ID INT PRIMARY KEY AUTO_INCREMENT,	
USUARIO VARCHAR(150) NOT NULL,    
PASSWORD VARCHAR(150) NOT NULL,
NOMBRE VARCHAR(150) NOT NULL,
APELLIDO VARCHAR(150) NOT NULL,
EMAIL VARCHAR(150) NOT NULL,
PERFIL VARCHAR(150) NOT NULL    
);

SELECT * FROM USUARIOS

CREATE TABLE COMPANIAS(
ID INT PRIMARY KEY AUTO_INCREMENT,
NOMBRE VARCHAR(150) NOT NULL,
DIRECCION VARCHAR(150) NOT NULL,
EMAIL VARCHAR(150) NOT NULL,
TELEFONO VARCHAR(150) NOT NULL,
CIUDAD VARCHAR(150) NOT NULL    
);

SELECT * FROM COMPANIAS

CREATE TABLE CIUDADES(
ID INT PRIMARY KEY AUTO_INCREMENT,
ID_PAIS INT NOT NULL,
NOMBRE_CIUDAD VARCHAR(150) NOT NULL  
);

SELECT * FROM CIUDADES 

CREATE TABLE PAISES(
ID INT PRIMARY KEY AUTO_INCREMENT,
ID_REGION INT NOT NULL,
NOMBRE_PAIS VARCHAR(150) NOT NULL  
);

SELECT * FROM PAISES

CREATE TABLE REGIONES(
ID INT PRIMARY KEY AUTO_INCREMENT,
NOMBRE_REGION VARCHAR(150) NOT NULL  
);

SELECT * FROM REGIONES
